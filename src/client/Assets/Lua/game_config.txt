local require = require
local print = print
local Resources = CS.UnityEngine.Resources
local dkjson = require "dkjson"
local pairs = pairs

module("game_config")

local gameConfig = nil

-- 读取Scripts\GameUpdate\Resources下的配置
local function load()
    local textAsset = Resources.Load("game_config")
    local config, pos, err = dkjson.decode (textAsset.text, 1, nil)
    if err then
        print ("Parsing game configuration with error:", err)
    else
        gameConfig = config
    end
end
load()

return gameConfig or {}
